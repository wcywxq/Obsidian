[509. 斐波那契数](https://leetcode.cn/problems/fibonacci-number)

```javascript
/**
 * 递归 - 内存溢出
 * @param {number} n
 * @return {number}
 */
var fib = function(n) {
  if (n === 0 || n === 1) return n
  return fib(n - 1) + fib(n - 2)
}

/**
 * 数组
 * @param {number} n
 * @return {number}
 */
var fib = function(n) {
  let arr = [0, 1]
  for (let i = 2; i <= n; i++) {
    arr.push(arr[i - 1] + arr[i - 2])
  }
  return arr[n]
}

/**
 * 迭代
 * @param {number} n
 * @return {number}
 */
var fib = function(n) {
  let a = 0
  let b = 1
  while (n--) {
    [a, b] = [b, a + b]
  }
  return a
}
```
