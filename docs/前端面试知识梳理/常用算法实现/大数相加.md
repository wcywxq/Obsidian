```javascript
function add(a, b) {
  let res = ''
  let temp = 0
  a = a.toString().split('')
  b = b.toString().split('')
  while (a.length || b.length || temp) {
      temp += ~~a.pop() + ~~b.pop()
      res += temp % 10
      temp = temp > 9
  }
  return res.replace(/^0+/, '')
}
```
