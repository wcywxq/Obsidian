## 利用递归的方法实现

```javascript
const flatDeep = arr => {
	let result = []
  for (let i = 0; i < arr.length; i++) {
  	if (arr[i] && arr[i].length) {
    	result = result.concat(flatDeep(arr[i]))
    } else {
    	result.push(arr[i])
    }
  }
  return result
}
```

## 利用 while + some 方法实现

```javascript
const flatDeep = arr => {
	while (arr.some(item => Array.isArray(item))) {
  	arr = [].concat(...arr)
  }
  return arr
}
```

## 利用 reduce + 递归 方法实现

```javascript
const flatDeep = arr => {
	return arr.reduce((prev, cur) => prev.concat(Array.isArray(cur) ? flatDeep(cur) : cur), [])
}
```
