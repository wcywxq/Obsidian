    let res = []
    let len = nums.length
    if (len < 3) return res

    nums.sort((a, b) => a - b)

    for (let i = 0; i < len - 2; i++) {

        let sum = nums[i] + nums[l] + nums[r]

          res.push([nums[i], nums[l], nums[r]])

          r--
        } else if (sum < 0) {
          l++
        } else {
          r--
        }
      }
    }
    return res
}
