var reverseVowels = function (s) {
  let vowels = 'aeiouAEIOU'
  let arr = Array.from(s)
  let l = 0
  let r = s.length - 1
  while (l < r) {
    let lIndex = vowels.indexOf(arr[l])
    let rIndex = vowels.indexOf(arr[r])
    if (lIndex === -1) {
      l++
    } else if (rIndex === -1) {
      r--
    } else {
      [arr[l], arr[r]] = [arr[r], arr[l]]
      l++
      r--
    }
  }
  return arr.join('')
}
