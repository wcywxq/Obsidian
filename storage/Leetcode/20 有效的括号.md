[20. 有效的括号](https://leetcode.cn/problems/valid-parentheses)

```javascript
/**
 * 维护栈
 * @param {string} s
 * @return {boolean}
 */
var isValid = function(s) {
    let map = new Map([
        ['(', ')'],
        ['{', '}'],
        ['[', ']']
    ])
    let stack = []
    for (let char of s) {
        if (map.has(char)) {
            stack.push(char)
        } else if (!stack.length || map.get(stack.pop()) !== char) {
            return false
        }
    }
    return !stack.length
}
```
