[46. 全排列](https://leetcode.cn/problems/permutations)

```javascript
/**
 * 回溯
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function(nums) {
  let res = []
  let path = []
  const dfs = () => {
    if (path.length === nums.length) {
      res.push([...path])
      return
    }
    for (let num of nums) {
      if (!path.includes(num)) {
        path.push(num)
        dfs()
        path.pop()
      }
    }
  }
  dfs()
  return res
}
```
