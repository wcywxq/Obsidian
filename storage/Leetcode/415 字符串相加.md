[415. 字符串相加](https://leetcode.cn/problems/add-strings)

```javascript
/**
 * 大数相加
 * @param {string} num1
 * @param {string} num2
 * @return {string}
 */
var addStrings = function(num1, num2) {
  let n1 = num1.toString().split('')
  let n2 = num2.toString().split('')
  let res = ''
  let temp = 0
  while (n1.length || n2.length || temp) {
    temp += ~~n1.pop() + ~~n2.pop()
    res += temp % 10
    temp = temp > 9
  }
  return res.split('').reverse().join('')
}
```
