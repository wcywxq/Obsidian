var canPlaceFlowers = function(flowerbed, n) {
  let i = 0
  let len = flowerbed.length
  while (i < len) {
    if (
      (i === 0 || flowerbed[i - 1] === 0) &&
      flowerbed[i] === 0 &&
      (i === len - 1 || flowerbed[i + 1] === 0)
    ) {
      n--
      flowerbed[i] = 1
    }
    i++
  }
  return n <= 0
}
