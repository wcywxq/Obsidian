/**

 * @param {number} target
 * @param {number[]} nums

var minSubArrayLen = function(target, nums) {
  let minLen = Number.MAX_SAFE_INTEGER
  let l = 0
  let r = 0
  let sum = 0
  while (r < nums.length) {
    sum += nums[r]
    while (sum >= target) {
      minLen = Math.min(minLen, r - l + 1)
      sum -= nums[l]
      l++
    }
    r++
  }
  return minLen === Number.MAX_SAFE_INTEGER ? 0 : minLen
}
