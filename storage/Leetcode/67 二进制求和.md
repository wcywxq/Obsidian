[67. 二进制求和](https://leetcode.cn/problems/add-binary)

```javascript
/**
 * 大数 bigInt 法
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var addBinary = function(a, b) {
  a = '0b' + a
  b = '0b' + b
  let sum = BigInt(a) + BigInt(b)
  return sum.toString(2)
}

/**
 * 利用位运算 + 二进制
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var addBinary = function(a, b) {
  // (a & b) << 1 得到进位数
  // a ^ b 不进位置加法，结果更新给 a
  // 得到进位数，更新给 b
  let n1 = BigInt('0b' + a)
  let n2 = BigInt('0b' + b)
  while (n2 !== 0n) {
    let curry = n1 & n2
    n1 = n1 ^ n2
    n2 = curry << 1n
  }
  return n1.toString(2)
}

/**
 * 字符串
 * @param {string} a
 * @param {string} b
 * @return {string}
 */
var addBinary = function(a, b) {
  let ans = ''
  let i = a.length - 1
  let j = b.length - 1
  let curry = 0
  while (i >= 0 || j >= 0) {
    let x = i >= 0 ? a[i--] - '0' : 0
    let y = j >= 0 ? b[j--] - '0' : 0
    let sum = x + y + curry
    ans += sum % 2
    curry = sum >> 1
  }
  if (curry) {
    ans += curry
  }
  return ans.split('').reverse().join('')
}
```
