## 递归

```javascript
function fn(n) {
	if (n === 0) return 0
  if (n === 1) return 1
  return fn(n - 2) + fn(n - 1)
}
```

## 尾调用

```javascript
function fn(n, a1 = 0, a2 = 1) {
	if (n === 0) return a1
  return fn(n - 1, a2, a1 + a2)
}
```

## 数组

```javascript
function fib(n) {
	const arr = [1, 1, 2]
  if (n <= arr.length) return arr[n]
  for (let i = arr.length; i < n; i++) {
  	arr.push(arr[i - 1] + arr[i - 2])
  }
  return arr[arr.length - 1]
}
```

## 迭代

```javascript
// method1
function fib(n) {
	let pre1 = 1
  let pre2 = 1
  let current = 2
  if (n <= 2) return current
  for (let i = 2; i < n; i++) {
  	pre1 = pre2
    pre2 = current
    current = pre1 + pre2
  }
  return current
}

// method2
function fib(n, a1 = 0, a2 = 1) {  
  while (n--) {
    [a1, a2] = [a2, a1 + a2]
  }
  return a1
}
```
